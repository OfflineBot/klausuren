\Sec{Welche Zerlegung wann? \hint{(Гњberblick)}}
\esel{Du willst ein MГ¶belstГјck transportieren вЂ” je nach Form nimmst du eine andere Zerlege-Strategie. Erst Form von $A$ anschauen, dann passende Zerlegung.}\par\vspace{1pt}
\begin{tabular}{@{}p{1.35cm}@{\,}p{1.6cm}@{\,}p{4.25cm}@{}}
\textbf{Zerleg.} & \textbf{Form $A$} & \textbf{wofГјr / Bild}\\[1pt]
$CR$ & bel.\ $m\times n$ & Rang, RГ¤ume вЂ” \emph{Fotoalbum}\\

\fld{Was ist am schnellsten} nimm die \emph{speziellste passende} Zerlegung (Aufwand Chol.\ $<LU<QR<$ SVD): symm.\ PD $\to$ \textbf{Cholesky}; quadr.\ normal $\to$ \textbf{$LU$}; schlecht kond./Ausgleich $\to$ \textbf{$QR$}; nicht quadr./NГ¤herung $\to$ \textbf{SVD}.\\
\fld{Existenz} SVD \& $CR$ \& $QR$: \emph{immer}. $LU$: quadr.\ \& fГјhrende Hauptminoren $\neq0$ (sonst $PA{=}LU$). Cholesky: nur symm.\ \emph{und} PD. Diagonalisierbar: nur mit $n$ unabh.\ EV (symm.\ $\Rightarrow$ immer).

\Sec{Format \& Matrix-Sichtweisen}
\fld{Format} $A$ ist $m\times n$: $m=$ Zeilen, $n=$ Spalten. $A\vec x$: $\vec x\in\mathbb R^n\to$ Ergebnis $\in\mathbb R^m$.\\
\esel{$A\vec x$ ist ein \emph{Rezept zum Mischen}: nimm $x_1$-mal Spalte 1 $+$ $x_2$-mal Spalte 2 $+\dots$}
\begin{tl}
\item $A\vec x$ = Linearkombination der \textbf{Spalten}; $\vec y\transp A$ = der \textbf{Zeilen}.
\item $\vec u\transp\vec v$ (inneres P., Zeile$\times$Spalte) = \textbf{Zahl}.
\item $\vec u\,\vec v\transp$ (Г¤uГџeres P., Spalte$\times$Zeile) = \textbf{Rang-1-Matrix}; $AB=\sum$ Rang-1.
\end{tl}
\textbf{Rang-1} $M=\vec u\,\vec v\transp$ (jede Zeile Vielfache der ersten): $\vec v\transp=$ erste Zeile, $\vec u=$ Faktoren je Zeile. Probe $\vec u\vec v\transp=M$.

% ============================================================
\Sec{Grundtechnik: LGS lГ¶sen mit GauГџ}
\esel{Treppe von oben nach unten sauber kehren: unter jeder Stufe (Pivot) den вЂћSchmutz'' zu $0$ machen.}
\fld{Ablauf}
\begin{nl}
\item \textbf{VorwГ¤rts:} unter der Diagonale Nullen. \merk{Faktor $=\dfrac{\text{Zahl die weg soll}}{\text{Zahl genau darГјber (Pivot)}}$} dann \emph{Zielzeile $-$ Faktor$\cdot$Pivotzeile}.
\item \textbf{RГјckwГ¤rts:} unterste Zeile zuerst (1 Variable), nach oben einsetzen.
\end{nl}
\textbf{Dreieckssysteme:} $L\vec y=\vec b$ vorwГ¤rts, $U\vec x=\vec y$ rГјckwГ¤rts.\\
\hint{$0\!=\!c\,(c\neq0)$: keine LГ¶sung. $0\!=\!0$: $\infty$ viele (freie Var.). Pivot $0$: Zeilen tauschen.}

\Sec{\dtag{CR}$A=CR$ \hint{(Fotoalbum)}}
\esel{Fotoalbum mit 3 Seiten, aber nur 2 echte Bilder hochgeladen вЂ” Seite 3 ist nur eine \emph{Collage} aus Seite 1\,\&\,2. $C=$ die echten Originalfotos (unabhГ¤ngige Spalten), $R=$ das \emph{Misch-Rezept}, wie man jede Spalte aus den Originalen zusammenrГјhrt.}
\fld{Geht} jede $m\times n$ (keine Voraussetzung). Nicht eindeutig (Pivotwahl).\\
\fld{Rezept} ($r=\#$Pivots)
\begin{nl}
\item $A$ auf RZSF bringen (Pivots $=1$, darГјber+darunter $0$).
\item $C=$ Pivot-Spalten aus \emph{original} $A$ (= Basis Spaltenraum).
\item $R=$ Nicht-Null-Zeilen der RZSF (= Misch-Rezept). Probe $CR=A$.
\end{nl}
\fld{Dimensionen} $\dim C(A)=\dim C(A\transp)=r$ (Zeilenrang $=$ Spaltenrang); $\dim N(A)=n-r$; $\dim N(A\transp)=m-r$ \hint{(Rangsatz $r+\dim N(A)=n$)}.\\
\fld{Beispiel} $A=\left(\begin{smallmatrix}1&3&5\\2&3&7\\1&3&5\end{smallmatrix}\right)$: Zeile 1\,$=$\,Zeile 3, Spalte 3 ist Mix $\Rightarrow r{=}2$. $C=\left(\begin{smallmatrix}1&3\\2&3\\1&3\end{smallmatrix}\right)$, $R=\left(\begin{smallmatrix}1&0&2\\0&1&1\end{smallmatrix}\right)$.\\
\hint{4 RГ¤ume: Zeilenraum $\perp N(A)$, Spaltenraum $\perp N(A\transp)$; $\mathbb R^n=C(A\transp)\oplus N(A)$.}

% ============================================================
\Sec{\dtag{LU\,=\,LR}$A=LU$ \hint{(Lego-Haus / Umzug)}}
\esel{MehrstГ¶ckiges Lego-Haus zum Umzug in 2 Teile zerlegen, einzeln transportieren, am Zielort wieder zusammenstecken. $L=$ Fundament (untere $\Delta$, $1$en auf Diag.), $U=R=$ Aufbau (obere $\Delta$). Ziel $A=L\cdot U$.}
\fld{Vorteil} viele rechte Seiten $\vec b$: nur \emph{einmal} zerlegen. Auch $\det$ \& Inverse.\\
\fld{Geht} nur \textbf{quadratisch} \& fГјhrende Hauptminoren $\neq0$. Pivot $0\to PA{=}LU$. $L$ eindeutig durch $1$-Diagonale.\\
\fld{Rezept}
\begin{nl}
\item GauГџ-VorwГ¤rts $\to U$. \merk{Faktor $=\frac{\text{Zahl die weg soll}}{\text{darГјber}}$}; obere Zeile$\cdot$Faktor von unterer abziehen.
\item Dieselben Faktoren (\emph{positiv!}) unter die Diagonale $\to L$.
\item LГ¶sen: $L\vec y=\vec b$ vorwГ¤rts, $U\vec x=\vec y$ rГјckwГ¤rts.
\end{nl}
\fld{Beispiel} $A=\left(\begin{smallmatrix}3&1&1\\6&3&4\\3&1&5\end{smallmatrix}\right)$: Faktoren $\tfrac63{=}2,\ \tfrac33{=}1,\ \tfrac{0}{\dots}$ $\Rightarrow L=\left(\begin{smallmatrix}1&0&0\\2&1&0\\1&0&1\end{smallmatrix}\right)$, $U=\left(\begin{smallmatrix}3&1&1\\0&1&2\\0&0&4\end{smallmatrix}\right)$.\\
$\det A=u_{11}\cdots u_{nn}$ \hint{($\det L=1$; bei $k$ Zeilentauschen $\times(-1)^k$)}. Inverse: $A\vec x_i=\vec e_i$ je Spalte.\\
\fld{Pivotisierung} Zeile tauschen $\to$ partiell $PA{=}LU$; \emph{bereits eingetragene} Faktoren in $L$ \textbf{mit-tauschen}! Spalte\,+\,Zeile tauschen $\to$ voll $PAQ{=}LU$.

% ============================================================
\Sec{\dtag{$LL\transp$}Cholesky \hint{(BrГјcken-TГњV)}}
\esel{HГ¤lt eine HolzbrГјcke das Gewicht? вЂћPositiv definit'' $=$ die BrГјcke sackt an \emph{keiner} Stelle ab. Wir prГјfen von links nach rechts StГјck fГјr StГјck вЂ” bestehen alle Checks, gibt's das \emph{TГњV-Siegel}.}
\fld{Geht} nur \textbf{symm.\ \& PD}. Eindeutig ($\ell_{jj}>0$). $\sim2\times$ schneller als $LU$; \emph{Existenz beweist PD}.\\
\Sub{Teil a) PD-Check \hint{(fГјhrende Hauptminoren $>0$)}}
\begin{tl}
\item Check 1: $a_{11}>0$? \quad Check 2: $\det$ der $2{\times}2$-Ecke $>0$?
\item Check 3: $\det A>0$? \ \emph{Ein} Check $\le0\Rightarrow$ BrГјcke bricht (nicht PD).
\end{tl}
\Sub{Teil b) Cholesky bauen \hint{(вЂћDetektivspiel'': $\ell_{ij}$ aus $A{=}LL\transp$ rГјckwГ¤rts puzzeln)}}
Reihenfolge $\ell_{11},\ell_{21},\ell_{31},\ell_{22},\ell_{32},\ell_{33}$:
\[\begin{array}{@{}l@{\ }l@{\ }l@{}}
\ell_{11}{=}\sqrt{a_{11}} & & \\[1pt]
\ell_{21}{=}\tfrac{a_{21}}{\ell_{11}} & \ell_{22}{=}\sqrt{a_{22}-\ell_{21}^2} & \\[1pt]
\ell_{31}{=}\tfrac{a_{31}}{\ell_{11}} & \ell_{32}{=}\tfrac{a_{32}-\ell_{31}\ell_{21}}{\ell_{22}} & \ell_{33}{=}\sqrt{a_{33}-\ell_{31}^2-\ell_{32}^2}
\end{array}\]
\hint{Diagonale: Wurzel aus вЂћ$a_{jj}-$ Quadrate links''. Darunter: durch $\ell_{jj}$ teilen вЂ” \emph{nicht} $a_{jj}$!} Radikand $\le0\Rightarrow$ nicht PD.\\
\fld{Beispiel} $A=\left(\begin{smallmatrix}16&4&8\\4&5&1\\8&1&13\end{smallmatrix}\right)$: $\ell_{11}{=}4,\ \ell_{21}{=}1,\ \ell_{31}{=}2,\ \ell_{22}{=}2,\ \ell_{32}{=}-\tfrac12,\ \ell_{33}{=}\sqrt{8{,}75}$.\\
\fld{allg.} $\ell_{jj}{=}\sqrt{a_{jj}-\sum_{k<j}\ell_{jk}^2}$,\ $\ell_{ij}{=}\big(a_{ij}-\sum_{k<j}\ell_{ik}\ell_{jk}\big)/\ell_{jj}$. LГ¶sen: $L\vec y{=}\vec b$, dann $L\transp\vec x{=}\vec y$.\\
\hint{Aus $A{=}LU$ direkt: $L_c=L\sqrt{D}$ mit $D=$ Diag.\ von $U$.}

\Sec{\dtag{QR}$A=QR$ \hint{(schiefes Foto geraderГјcken)}}
\esel{Du fotografierst ein Dokument, hГ¤ltst das Handy aber schief. $Q$ \emph{dreht} das Bild gerade (orthogonale Rotation, Г¤ndert nichts am Inhalt), $R$ ist das saubere, gerade Bild (obere $\Delta$, leicht lesbar). $A=Q\cdot R$.}
\fld{Eigenschaften} $Q$ orthonormale Spalten ($Q\transp Q{=}E$, $Q\inv{=}Q\transp$), $R$ obere $\Delta$. Sehr \emph{stabil}.\\
\fld{Geht} jede $m\times n$; bei vollem Rang \& $r_{ii}>0$ eindeutig.\\
\fld{$Q\inv$ ablesen} weil orthogonal: $Q\inv=Q\transp$ вЂ” einfach 1.\ Zeile als 1.\ Spalte usw.\ \hint{z.B.\ $Q{=}\left(\begin{smallmatrix}5/13&-12/13\\12/13&5/13\end{smallmatrix}\right)\Rightarrow Q\inv{=}\left(\begin{smallmatrix}5/13&12/13\\-12/13&5/13\end{smallmatrix}\right)$.}\\
\fld{Gram-Schmidt} $\tilde q_k=\vec v_k-\sum_{i<k}\frac{\tilde q_i\transp\vec v_k}{\|\tilde q_i\|^2}\tilde q_i$, dann $\hat q_k=\tilde q_k/\|\tilde q_k\|$; $R=Q\transp A$ \hint{($r_{ii}{=}\|\tilde q_i\|$ auf Diag.)}.\\
\fld{LGS mit QR \hint{(VIP-Pass)}} \esel{Statt das schwere $A\vec x{=}\vec b$ direkt zu lГ¶sen, nimmst du den VIP-Hintereingang durch 2 TГјren $Q$ dann $R$.}
\[ R\vec x=Q\transp\vec b\quad\text{(nur noch rГјckwГ¤rts einsetzen)} \]
\hint{Schritt 1: $\vec d=Q\transp\vec b$ (Zeile$\cdot$Spalte). Schritt 2: $R\vec x=\vec d$ rГјckwГ¤rts. Spart bei Regression: $X\transp X=R\transp R$.}

\Sec{\dtag{$S\Lambda S\inv$}Diagonalisieren \hint{(Regal im Umzugskarton)}}
\esel{Das Regal passt nicht ins Auto. LГ¶sung: in Einzelteile zerlegen, im Karton transportieren, am Ziel in \emph{selber Reihenfolge} wieder zusammenbauen. $\Lambda=$ Umzugskarton (Eigenwerte sortiert auf der Diagonale), $S=$ Aufbau-Anleitung (Eigenvektoren nebeneinander). \fld{Vorteil} $A^k=S\Lambda^k S\inv$, $A\inv=S\Lambda\inv S\inv$ (Kehrwerte der $\lambda$); Definitheit aus den $\lambda$.\\
\fld{Geht} quadr.\ \& $n$ unabh.\ EV вЂ” \emph{nicht} jede Matrix! Symm.\ $\Rightarrow$ \emph{immer} (Spektralsatz: reelle $\lambda$, orthogonale EV, $S\inv=Q\transp$).\\
\fld{Rezept}
\begin{nl}
\item $\lambda_i$ aus $\det(A-\lambda E)=0$.
\item EV: $(A-\lambda_i E)\vec v=\vec0$, eine Komponente frei.
\item $S=$ EV als Spalten (symm.: normieren $\to Q$), $\Lambda=\mathrm{diag}(\lambda_i)$.
\end{nl}

\Sec{\dtag{SVD}$A=U\Sigma V\transp$ \hint{(DrehenвЂ“StreckenвЂ“Drehen)}}
\esel{Jede Matrix macht 3 Dinge nacheinander: erst \emph{drehen} ($V\transp$), dann entlang der Achsen \emph{strecken} ($\Sigma$), dann nochmal \emph{drehen} ($U$). Funktioniert fГјr \emph{jede} Matrix вЂ” auch nicht-quadratisch.}
\fld{Bausteine} $V$ ($n{\times}n$ orth., EV von $A\transp A$); $\Sigma$ ($m{\times}n$, $\sigma_1\ge\dots\ge\sigma_r>0$ auf Diag.\ \hint{(nicht quadr.\ falls $m\neq n$ вЂ” mit Nullen auffГјllen)}); $U$ ($m{\times}m$ orth., EV von $AA\transp$).\\
\fld{Rezept} \merk{$A$ nicht quadratisch? Erst вЂћkГјnstlich quadratisch'': $A\transp A$ ($=$ Kippen mal Original).}
\begin{nl}
\item $A\transp A$: EW $\lambda_i\ge0$, norm.\ EV $\to$ Spalten von $V$. \hint{($A\transp A$ \emph{quadriert} die Werte.)}
\item $\sigma_i=\sqrt{\lambda_i}$ absteigend \hint{(Wurzel ziehen $=$ Gegenteil von Quadrieren)} $\to\Sigma$.
\item $\vec u_i=\frac{1}{\sigma_i}A\vec v_i\to$ Spalten von $U$ (nur $\sigma_i\neq0$); Rest mit Gram-Schmidt/Kreuzprodukt zu ONB.
\end{nl}
\hint{Kleinere von $A\transp A$ / $AA\transp$ wГ¤hlen! Symm.\ PD $\Rightarrow$ SVD $=$ Eigenzerlegung, $\sigma_i=|\lambda_i|$.}\\
\fld{Liefert} Rang $=\#\{\sigma_i>0\}$; reduziert $A=U_r\Sigma_r V_r\transp$ (Nullzeilen/-spalten weg); beste Rang-$k$-NГ¤herung (Eckart-Young); $A^{+}$; PCA.

\fld{Nutzen} Гјberbestimmt $\to$ kleinster Fehler $\|A\vec x-\vec b\|$; unterbestimmt $\to$ kleinste Norm $\|\vec x\|$.

\Sec{Eigenwerte \& Eigenvektoren}
$A\vec v=\lambda\vec v$ вЂ” Richtung bleibt, wird nur um $\lambda$ gestreckt.\\
\fld{Dreiecksregel} $\Delta$-Matrix (eine Ecke nur Nullen) $\Rightarrow$ EW stehen \emph{direkt} auf der Diagonale. \hint{Auch вЂћverstecktes'' Dreieck per Ausblenden einer Zeile/Spalte.}\\
\fld{Sonst: $\det(A-\lambda E)=0$} \esel{Waage mit Paket: du willst den Streckungsfaktor (das вЂћPaket'' $\lambda$) wissen вЂ” also ziehst du $\lambda$ von der Diagonale ab und schaust, was bleibt.}\\
$2{\times}2$: $\lambda=\tfrac{\mathrm{tr}\pm\sqrt{\mathrm{tr}^2-4\det}}{2}$.\\
\merk{Ausklammer-Trick} $(a-\lambda)(d-\lambda)$: das вЂћ$-\lambda$'' gehГ¶rt wie ein \emph{Rucksack} fest zur Zahl. Wie вЂћ4 BegrГјГџungen'': jeder aus Klammer 1 schГјttelt jedem aus Klammer 2 die Hand.\\
\fld{Eigenvektor \hint{(RГ¤tsel)}} $\lambda$ einsetzen, $(A-\lambda E)\vec v=\vec0$ lГ¶sen: eine Komponente \textbf{frei} wГ¤hlen ($v_1{=}1$), Rest folgt. \hint{Nie $\vec v=\vec0$!}\\
\fld{Beispiel} $A=\left(\begin{smallmatrix}5&2\\2&2\end{smallmatrix}\right)$: $\lambda^2-7\lambda+6=0\Rightarrow\lambda{=}6,1$; $\vec v_6{=}\left(\begin{smallmatrix}2\\1\end{smallmatrix}\right),\ \vec v_1{=}\left(\begin{smallmatrix}-1\\2\end{smallmatrix}\right)$.\\
\fld{EW-Rechenregeln} $A+\gamma E$ hat EW $\lambda+\gamma$ (gleiche EV); $A\inv$ hat $1/\lambda$; $A^k$ hat $\lambda^k$. \hint{Kontrolle: $\sum\lambda_i{=}\mathrm{tr}$, $\prod\lambda_i{=}\det$.}

% ============================================================
\Sec{Determinante, Inverse, Spur}
\fld{Det $2{\times}2$} $\det\big(\begin{smallmatrix}a&b\\c&d\end{smallmatrix}\big)=ad-bc$.\\
\fld{Det $3{\times}3$} (Sarrus) $a(ei{-}fh)-b(di{-}fg)+c(dh{-}eg)$.\\
\fld{Inverse $2{\times}2$} $\big(\begin{smallmatrix}a&b\\c&d\end{smallmatrix}\big)\inv=\frac{1}{ad-bc}\big(\begin{smallmatrix}d&-b\\-c&a\end{smallmatrix}\big)$ \hint{(Diag.\ tauschen, Neben-Diag.\ Vorzeichen).}\\
\fld{Inverse allg.} GauГџ-Jordan $(A\,|\,E)\to(E\,|\,A\inv)$, oder $A\vec x_i=\vec e_i$ spaltenweise.\\
\fld{Spur} $\mathrm{tr}(A)=\sum_i a_{ii}=\sum\lambda_i$. \merk{Zyklisch} $\mathrm{tr}(ABC)=\mathrm{tr}(BCA)=\mathrm{tr}(CAB)$ \hint{(nur zyklisch drehen, nicht beliebig!)}.

% ============================================================
\Sec{Definitheit \hint{(symm.\ $A$ вЂ” BrГјcken-TГњV)}}
\esel{Wie bei Cholesky: PD $=$ die вЂћBrГјcke'' sackt nirgends ab ($\vec x\transp A\vec x>0\ \forall\vec x\neq0$).}
\fld{Test 1 (EW)}
\begin{tl}
\item \textbf{PD:} alle $\lambda_i>0$;\quad \textbf{PSD:} $\ge0$;\quad \textbf{ND:} $<0$;\quad \textbf{NSD:} $\le0$
